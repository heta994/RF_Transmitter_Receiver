RF Transmitter & Receiver Project

Project Description

This Arduino project demonstrates sending and receiving messages using RF modules with the RadioHead library. The transmitter sends a "Hello World!" message every second, and the receiver prints it on the Serial Monitor.

Requirements

  • Arduino Board (Uno, Nano, etc.)
  • 433MHz RF Transmitter & Receiver modules
  • Arduino IDE with RadioHead library installed

Transmitter Code


#include <RH_ASK.h>
#include <SPI.h> // Not actually used but needed

RH_ASK driver;

void setup() {
    Serial.begin(9600);
    if (!driver.init()) {
        Serial.println("init failed");
    }
}

void loop() {
    const char *msg = "Hello World!";
    driver.send((uint8_t *)msg, strlen(msg));
    driver.waitPacketSent();
    Serial.println("Message sent: Hello World!");
    delay(1000);
}

Receiver Code


#include <RH_ASK.h>
#include <SPI.h>

RH_ASK driver;

void setup() {
    Serial.begin(9600);
    if (!driver.init()) {
        Serial.println("init failed");
    }
}

void loop() {
    uint8_t buf[12];
    uint8_t buflen = sizeof(buf);
    if (driver.recv(buf, &buflen)) {
        Serial.print("Message received: ");
        Serial.println((char*)buf);
    }
}
